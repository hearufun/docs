## Installation

```bash
npm install @talkos/sdk
# or
yarn add @talkos/sdk
# or
pnpm add @talkos/sdk
```

**Requirements:** Node.js 18+ or any modern browser

---

## Initialization

```javascript
import TalkOS from '@talkos/sdk';

const client = new TalkOS({
  apiKey: process.env.TALKOS_API_KEY,
  tenantId: process.env.TALKOS_TENANT_ID,
  baseUrl: 'https://api.talkos.io'  // optional, defaults to production
});
```

## Calls

### Initiate a Call

```javascript
const call = await client.calls.initiate({
  to: '+14155551234',
  systemPrompt: 'You are a helpful customer support agent.',
  voice: { voiceId: 'rachel', speed: 1.0 },
  maxDuration: 300,
  webhookUrl: 'https://your-app.com/webhook',
  metadata: { customerId: 'cust-001' }
});

console.log(call.callUuid);  // "call-uuid-123"
console.log(call.status);    // "queued"
```

### Batch Calls

```javascript
const batch = await client.calls.batch({
  calls: [
    { to: '+14155551001', systemPrompt: '...' },
    { to: '+14155551002', systemPrompt: '...' },
  ],
  concurrency: 5,
  retryPolicy: { maxRetries: 2, retryOn: ['no_answer'] }
});

console.log(batch.batchId);
console.log(batch.totalCalls);
```

### Get Call Status

```javascript
const status = await client.calls.get('call-uuid-123');
console.log(status.state);     // "in-progress" | "completed" | "failed"
console.log(status.duration);   // 145
console.log(status.sentiment);  // 0.7
```

### Hang Up

```javascript
await client.calls.hangup('call-uuid-123');
```

### List Calls

```javascript
const calls = await client.calls.list({
  dateFrom: '2026-01-01',
  dateTo: '2026-02-18',
  limit: 50
});
```

## Web Sessions

### Create a Session

```javascript
const session = await client.sessions.create({
  systemPrompt: 'You are a helpful assistant.',
  voice: { voiceId: 'rachel' },
  language: 'en',
  enableVideo: false
});
```

### Connect via WebSocket

```javascript
session.connect();

// Listen for events
session.on('connected', () => console.log('Session started'));
session.on('transcript', (data) => {
  console.log(`[${data.role}]: ${data.text}`);
});
session.on('audio', (chunk) => {
  // Play audio chunk in browser
  audioPlayer.play(chunk);
});
session.on('ended', (data) => {
  console.log('Session ended:', data.reason);
});
session.on('error', (err) => console.error(err));

// Send audio from microphone
navigator.mediaDevices.getUserMedia({ audio: true })
  .then(stream => {
    const recorder = new MediaRecorder(stream);
    recorder.ondataavailable = (e) => session.sendAudio(e.data);
    recorder.start(100);  // Send chunks every 100ms
  });
```

### Control Session

```javascript
await session.pause();
await session.resume();
await session.updateConfig({ voice: { speed: 1.2 } });
await session.end();
```

### Get Transcript

```javascript
const transcript = await client.sessions.getTranscript('session-id');
```

## TypeScript Support

The SDK ships with full TypeScript definitions:

```typescript
import TalkOS, { 
  Call, 
  Session, 
  Tenant, 
  AnalyticsResponse,
  CallInitiateOptions 
} from '@talkos/sdk';

const options: CallInitiateOptions = {
  to: '+14155551234',
  systemPrompt: 'You are helpful.',
  maxDuration: 300
};

const call: Call = await client.calls.initiate(options);
```
