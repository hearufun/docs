---
title: "Import Number Inventory"
api: "POST https://api.talkos.io/api/numbers/inventory/import"
description: "Bulk import phone numbers into your inventory for rapid provisioning"
---

Import a batch of phone numbers into your tenant's inventory. This is useful for pre-purchasing numbers in bulk or importing numbers from another provider. Numbers in inventory can be quickly assigned to campaigns or agents without the delay of purchasing.

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your TalkOS API key
</ParamField>

<ParamField header="x-tenant-id" type="string" required>
  Your tenant identifier
</ParamField>

## Request Body

<ParamField body="numbers" type="array" required>
  Array of phone numbers to import
  <Expandable title="number object">
    <ParamField body="number" type="string" required>
      Phone number in E.164 format (e.g., `+14155551234`)
    </ParamField>
    <ParamField body="dialerProvider" type="string" required>
      The dialer provider this number belongs to (`plivo`, `exotel`)
    </ParamField>
    <ParamField body="providerNumberId" type="string">
      Provider's unique identifier for the number
    </ParamField>
    <ParamField body="country" type="string" required>
      ISO country code
    </ParamField>
    <ParamField body="type" type="string" default="local">
      Number type (`local`, `toll-free`, `mobile`)
    </ParamField>
    <ParamField body="friendlyName" type="string">
      Friendly name or label
    </ParamField>
    <ParamField body="tags" type="object">
      Custom key-value tags
    </ParamField>
    <ParamField body="capabilities" type="object">
      Number capabilities (voice, sms, mms)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="source" type="string">
  Source of the import (e.g., `provider-migration`, `bulk-purchase`, `manual`)
</ParamField>

<ParamField body="defaultStatus" type="string" default="available">
  Default status for imported numbers
  - `available` - Ready to be assigned
  - `reserved` - Reserved but not yet assigned
  - `quarantine` - Under review before use
</ParamField>

<ParamField body="validateWithProvider" type="boolean" default="false">
  Validate each number with the provider before importing. Slower but ensures accuracy.
</ParamField>

<ParamField body="skipDuplicates" type="boolean" default="true">
  Skip numbers that already exist in inventory instead of failing
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="importId" type="string">
      Unique identifier for this import batch
    </ResponseField>
    <ResponseField name="status" type="string">
      Import status (completed, partial, failed)
    </ResponseField>
    <ResponseField name="summary" type="object">
      <Expandable title="summary">
        <ResponseField name="total" type="integer">
          Total numbers in request
        </ResponseField>
        <ResponseField name="imported" type="integer">
          Successfully imported
        </ResponseField>
        <ResponseField name="skipped" type="integer">
          Skipped (duplicates)
        </ResponseField>
        <ResponseField name="failed" type="integer">
          Failed to import
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="imported" type="array">
      Array of successfully imported number IDs
    </ResponseField>
    <ResponseField name="errors" type="array">
      Array of error details for failed imports
      <Expandable title="error object">
        <ResponseField name="number" type="string">
          The phone number that failed
        </ResponseField>
        <ResponseField name="code" type="string">
          Error code
        </ResponseField>
        <ResponseField name="message" type="string">
          Error message
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="importedAt" type="string">
      Import completion timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

## Limits

- Maximum 100 numbers per import request
- For larger imports, use multiple requests or contact support for bulk import options
