---
title: "List Number Inventory"
api: "GET https://api.talkos.io/api/numbers/inventory"
description: "Retrieve all phone numbers in your tenant's inventory"
---

Get a paginated list of all phone numbers in your inventory. The inventory contains numbers that are available for assignment, reserved, or in quarantine.

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your TalkOS API key
</ParamField>

<ParamField header="x-tenant-id" type="string" required>
  Your tenant identifier
</ParamField>

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="50">
  Number of results per page (1-100)
</ParamField>

<ParamField query="status" type="string">
  Filter by inventory status
  - `available` - Ready for assignment
  - `reserved` - Reserved for specific use
  - `assigned` - Currently assigned to a service
  - `quarantine` - Under review
</ParamField>

<ParamField query="country" type="string">
  Filter by country code (e.g., `US`, `IN`)
</ParamField>

<ParamField query="type" type="string">
  Filter by number type (`local`, `toll-free`, `mobile`)
</ParamField>

<ParamField query="dialerProvider" type="string">
  Filter by dialer provider (`plivo`, `exotel`)
</ParamField>

<ParamField query="search" type="string">
  Search by number or friendly name
</ParamField>

<ParamField query="tags" type="string">
  Filter by tags (comma-separated key:value pairs, e.g., `campaign:q1,region:west`)
</ParamField>

<ParamField query="importId" type="string">
  Filter by import batch ID
</ParamField>

<ParamField query="sortBy" type="string" default="createdAt">
  Sort field (`number`, `status`, `createdAt`, `lastAssignedAt`)
</ParamField>

<ParamField query="sortOrder" type="string" default="desc">
  Sort order (`asc`, `desc`)
</ParamField>

```bash cURL
curl -X GET "https://api.talkos.io/api/numbers/inventory?status=available&country=US&limit=20" \
  -H "x-api-key: your-api-key" \
  -H "x-tenant-id: your-tenant-id"
```

```javascript Node.js
const response = await fetch(
  'https://api.talkos.io/api/numbers/inventory?status=available&country=US&limit=20',
  {
    method: 'GET',
    headers: {
      'x-api-key': 'your-api-key',
      'x-tenant-id': 'your-tenant-id'
    }
  }
);

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.talkos.io/api/numbers/inventory',
    params={
        'status': 'available',
        'country': 'US',
        'limit': 20
    },
    headers={
        'x-api-key': 'your-api-key',
        'x-tenant-id': 'your-tenant-id'
    }
)

data = response.json()
```
