---
title: "List Phone Numbers"
api: "GET https://api.talkos.io/api/numbers"
description: "Retrieve all phone numbers associated with your tenant"
---

Get a paginated list of all phone numbers owned by your tenant. Supports filtering by various criteria and includes full details for each number.

## Authentication

<ParamField header="x-api-key" type="string" required>
  Your TalkOS API key
</ParamField>

<ParamField header="x-tenant-id" type="string" required>
  Your tenant identifier
</ParamField>

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of results per page (1-100)
</ParamField>

<ParamField query="country" type="string">
  Filter by country code (e.g., `US`, `IN`)
</ParamField>

<ParamField query="type" type="string">
  Filter by number type
  - `local` - Local numbers
  - `toll-free` - Toll-free numbers
  - `mobile` - Mobile numbers
</ParamField>

<ParamField query="direction" type="string">
  Filter by call direction configuration
  - `outbound` - Outbound only
  - `inbound` - Inbound only
  - `both` - Both directions
</ParamField>

<ParamField query="status" type="string">
  Filter by number status
  - `active` - Active numbers
  - `pending` - Pending activation
  - `suspended` - Suspended numbers
</ParamField>

<ParamField query="dialerProvider" type="string">
  Filter by dialer provider (`plivo`, `exotel`)
</ParamField>

<ParamField query="search" type="string">
  Search by number or friendly name
</ParamField>

<ParamField query="sortBy" type="string" default="createdAt">
  Sort field (`number`, `friendlyName`, `createdAt`, `status`)
</ParamField>

<ParamField query="sortOrder" type="string" default="desc">
  Sort order (`asc`, `desc`)
</ParamField>


```javascript Node.js
const response = await fetch(
  'https://api.talkos.io/api/numbers?page=1&limit=20&status=active',

    method: 'GET',
    headers: {

      'x-tenant-id': 'your-tenant-id'
    }
  }
);
```

```python Python
response = requests.get(

    },
    headers={
        'x-api-key': 'your-api-key',
```
