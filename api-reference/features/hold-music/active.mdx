---
title: 'Get Active Holds'
api: 'GET https://api.talkos.io/api/features/hold-music/active'
description: 'Get all currently active hold sessions'
---

Retrieve a list of all calls currently on hold with hold music playing.

## Query Parameters

<ParamField query="limit" type="number" default={50}>
  Maximum number of results to return.
</ParamField>

<ParamField query="offset" type="number" default={0}>
  Number of results to skip for pagination.
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Whether the request was successful.
</ResponseField>

<ResponseField name="holds" type="array">
  Array of active hold sessions.
  <Expandable title="hold properties">
    <ResponseField name="holdId" type="string">
      Unique identifier for the hold session.
    </ResponseField>
    <ResponseField name="callUuid" type="string">
      The call's unique identifier.
    </ResponseField>
    <ResponseField name="callerNumber" type="string">
      The caller's phone number (masked if masking is enabled).
    </ResponseField>
    <ResponseField name="track" type="object">
      Information about the track being played.
    </ResponseField>
    <ResponseField name="startedAt" type="string">
      ISO 8601 timestamp when hold started.
    </ResponseField>
    <ResponseField name="duration" type="number">
      Current hold duration in seconds.
    </ResponseField>
    <ResponseField name="queuePosition" type="number">
      Caller's position in the queue (if applicable).
    </ResponseField>
    <ResponseField name="estimatedWaitTime" type="number">
      Estimated remaining wait time in seconds.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="totalCount" type="number">
  Total number of active holds.
</ResponseField>

<ResponseField name="averageHoldTime" type="number">
  Average hold duration across all active holds in seconds.
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.talkos.io/api/features/hold-music/active?limit=20" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://api.talkos.io/api/features/hold-music/active?limit=20', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.talkos.io/api/features/hold-music/active',
    headers={
        'Authorization': 'Bearer YOUR_API_KEY'
    },
    params={
        'limit': 20
    }
)

data = response.json()
```
