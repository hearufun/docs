---
title: "List Knowledge Bases"
api: "GET https://api.talkos.io/api/knowledge-base"
description: "Retrieve all knowledge bases for your tenant"
---

## Overview

List all knowledge bases associated with your tenant. Returns summary information for each knowledge base including document counts and total indexed tokens.

## Headers

<ParamField header="x-api-key" type="string" required>
  Your TalkOS API key
</ParamField>

<ParamField header="x-tenant-id" type="string" required>
  Your tenant identifier
</ParamField>

## Query Parameters

<ParamField query="page" type="number">
  Page number for pagination. Default: `1`.
</ParamField>

<ParamField query="limit" type="number">
  Number of results per page. Default: `20`. Maximum: `100`.
</ParamField>

<ParamField query="sortBy" type="string">
  Field to sort by. Options: `name`, `createdAt`, `documentCount`. Default: `createdAt`.
</ParamField>

<ParamField query="sortOrder" type="string">
  Sort direction. Options: `asc`, `desc`. Default: `desc`.
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request completed successfully
</ResponseField>

<ResponseField name="knowledgeBases" type="array">
  Array of knowledge base objects
  <Expandable title="knowledge base object">
    <ResponseField name="id" type="string">
      Unique identifier for the knowledge base
    </ResponseField>
    <ResponseField name="name" type="string">
      Name of the knowledge base
    </ResponseField>
    <ResponseField name="description" type="string">
      Description of the knowledge base
    </ResponseField>
    <ResponseField name="embeddingModel" type="string">
      The embedding model used for vectorization
    </ResponseField>
    <ResponseField name="documentCount" type="number">
      Number of documents in the knowledge base
    </ResponseField>
    <ResponseField name="totalTokens" type="number">
      Total tokens indexed across all documents
    </ResponseField>
    <ResponseField name="lastUpdatedAt" type="string">
      ISO 8601 timestamp of last document upload
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  <Expandable title="properties">
    <ResponseField name="page" type="number">
      Current page number
    </ResponseField>
    <ResponseField name="limit" type="number">
      Results per page
    </ResponseField>
    <ResponseField name="totalPages" type="number">
      Total number of pages
    </ResponseField>
    <ResponseField name="totalCount" type="number">
      Total number of knowledge bases
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.talkos.io/api/knowledge-base" \
  -H "x-api-key: your-api-key" \
  -H "x-tenant-id: your-tenant-id"
```

```bash cURL (With Pagination)
curl -X GET "https://api.talkos.io/api/knowledge-base?page=1&limit=10&sortBy=name&sortOrder=asc" \
  -H "x-api-key: your-api-key" \
  -H "x-tenant-id: your-tenant-id"
```
</RequestExample>
